                  Sub Kronk()'' https://i.imgur.com/znG2VQI.jpg'Call START_P1Call START_P2Call START_P3Call START_P4Call START_P5Call START_P5Call START_P5End SubSub START_P1()'' REMOVE Top Row, Columns''    Rows("1:1").Select    Selection.Delete shift:=xlUp    Columns("A:A").Select    Selection.Delete shift:=xlToLeft    Columns("E:E").Select    Selection.Delete shift:=xlToLeft    Selection.Delete shift:=xlToLeft    Selection.Delete shift:=xlToLeft    Columns("F:F").Select    Selection.Delete shift:=xlToLeftEnd SubSub START_P2()'' DELETE uncolored rows''    Cells.Select    Selection.AutoFilter    ActiveSheet.Range("$A$1:$R$65").AutoFilter Field:=1, Operator:= _        xlFilterNoFill    Selection.ClearContentsEnd SubSub START_P3()'' Sort by TIME, then COLLEGE, then LAST NAME''    Cells.Select    ActiveWorkbook.ActiveSheet.Sort.SortFields.Clear    ActiveWorkbook.ActiveSheet.Sort.SortFields.Add2 Key:=Range("E1:E64") _        , SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal    ActiveWorkbook.ActiveSheet.Sort.SortFields.Add2 Key:=Range("G1:G64") _        , SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal    ActiveWorkbook.ActiveSheet.Sort.SortFields.Add2 Key:=Range("A1:A64") _        , SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal    With ActiveWorkbook.ActiveSheet.Sort        .SetRange Range("A1:J64")        .Header = xlGuess        .MatchCase = False        .Orientation = xlTopToBottom        .SortMethod = xlPinYin        .Apply    End WithEnd SubSub START_P4()'' Remove COLOR''    Cells.Select    With Selection.Interior        .Pattern = xlNone        .TintAndShade = 0        .PatternTintAndShade = 0    End WithEnd SubSub START_P5()'Dim iRow As Integer, iCol As IntegerDim oRng As RangeSet oRng = Range("g1")iRow = oRng.RowiCol = oRng.ColumnDo'If Cells(iRow + 1, iCol) <> Cells(iRow,
